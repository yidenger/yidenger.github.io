Mysql复习

所谓温故而知新，好记性比不过烂笔头，所以复习是很有必要的。

Mysql登录

u表示用户名，小写p表示密码，大写P表示端口号，h表示服务器（即IP），大写D表示数据库

创建数据库

create database test1; -- 创建数据库test1；

修改数据库字符编码

alter database test1 character set = utf8;

删除数据库

drop database test1;

创建表

create table user(
    userId int not null auto_increment comment '用户id',
    username varchar(30) not null comment '用户名',
    `password` varchar(30) not null comment '密码',
    loginTime timestamp not null default current_timestamp on update current_timestamp comment '登录时间',
    primary key(userId)
)comment '用户表';

查看表结构

show columns from table_name;

修改数据表

-- 添加列
alter table user add age tinyint comment '年龄';
-- 修改列的属性
alter table user modify age smallint;
-- 修改列名
alter table user change age ages smallint;
-- 删除列
alter table user drop ages;

插入表

-- 添加用户（往user表中插入数据）
insert into user (username, `password`) values ('admin', '123456');
-- 使用set插入数据
insert into user set username = 'people1', `password` = '567890';
-- 从其他表数据插入
insert into user (username, `password`) select (username, `password`) from user_tmp;

更新表

-- 修改密码(更新password字段)
update user set `password` = '66666' where userid = 2;
-- 更新age字段
update user set age = age +1;

删除记录

-- 删除用户
delete from user where userid = 2;

Mysql修改表、字段的字符集、字符序

-- 修改数据库的字符集
alter database db_name default character set character_name [collate...];

-- 把表默认的字符集合所有字符列（char、varchar、text）改为新的字符集
-- 格式：alter table table_name convert to character set character_name [collate...];
alter table user convert to character set utf8 collate utf8_unicode_ci;

-- 只是修改表的默认字符集
-- alter table tb_name default character set character_name [collate...];
alter table user default character set utf8 collate utf8_unicode_ci;

-- 修改字段的字符集
-- alter table tb_name change c_name c_name character set char_name [collate...];
alter table user change username username character set utf8 collate utf8_unicode_ci;

--查看字段编码
show full columns from user;
show variables like '%colla%';
